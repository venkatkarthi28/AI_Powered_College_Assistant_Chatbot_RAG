const mongoose = require('mongoose');

/**
 * Contact Schema — core entity for the People module
 * Supports multi-tenancy via orgId
 */
const contactSchema = new mongoose.Schema(
  {
    orgId: {
      type: mongoose.Schema.Types.ObjectId,
      required: true,
      index: true,
    },

    firstName: { type: String, required: true, trim: true },
    lastName:  { type: String, default: '',   trim: true },
    email:     { type: String, required: true, lowercase: true, trim: true },
    phone:     { type: String, default: '' },
    avatarInitials: { type: String },

    tags:    { type: [String], default: [] },
    listIds: [{ type: mongoose.Schema.Types.ObjectId, ref: 'ContactList' }],

    isVip:        { type: Boolean, default: false },
    isSubscribed: { type: Boolean, default: true },
    emailStatus:  { type: String, enum: ['valid', 'invalid', 'unknown'], default: 'valid' },

    source:   { type: String, enum: ['manual', 'csv', 'api'], default: 'manual' },
    company:  { type: String, default: '' },
    location: { type: String, default: '' },
    notes:    { type: String, default: '' },
  },
  { timestamps: true }
);

// One email per org
contactSchema.index({ orgId: 1, email: 1 }, { unique: true });

// Auto-generate avatar initials on save
contactSchema.pre('save', function (next) {
  if (this.isModified('firstName') || this.isModified('lastName')) {
    const f = (this.firstName || '').charAt(0).toUpperCase();
    const l = (this.lastName  || '').charAt(0).toUpperCase();
    this.avatarInitials = f + l || f;
  }
  next();
});

module.exports = mongoose.model('Contact', contactSchema);

const Contact     = require('../models/Contact');
const ContactList = require('../models/ContactList');
const mongoose    = require('mongoose');

const toObjId = (id) => new mongoose.Types.ObjectId(id);

// ── SUMMARY ──────────────────────────────────────────────
exports.getSummary = async (req, res) => {
  try {
    const orgId = toObjId(req.user.orgId);

    const [totals, listCount] = await Promise.all([
      Contact.aggregate([
        { $match: { orgId } },
        {
          $group: {
            _id: null,
            total:      { $sum: 1 },
            vip:        { $sum: { $cond: ['$isVip', 1, 0] } },
            validEmail: { $sum: { $cond: [{ $eq: ['$emailStatus', 'valid'] }, 1, 0] } },
          },
        },
      ]),
      ContactList.countDocuments({ orgId }),
    ]);

    const startOfMonth = new Date();
    startOfMonth.setDate(1);
    startOfMonth.setHours(0, 0, 0, 0);
    const newThisMonth = await Contact.countDocuments({ orgId, createdAt: { $gte: startOfMonth } });

    const t = totals[0] || { total: 0, vip: 0, validEmail: 0 };

    res.json({
      success: true,
      data: {
        totalContacts:  t.total,
        vipCount:       t.vip,
        listCount,
        validEmailRate: t.total ? Math.round((t.validEmail / t.total) * 100) : 0,
        newThisMonth,
      },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── GET ALL CONTACTS ──────────────────────────────────────
exports.getContacts = async (req, res) => {
  try {
    const orgId = toObjId(req.user.orgId);
    const { page = 1, limit = 15, search, tag, listId, tab = 'all' } = req.query;

    const match = { orgId };

    if (tab === 'tagged')  match.tags    = { $exists: true, $not: { $size: 0 } };
    if (tab === 'lists')   match.listIds = { $exists: true, $not: { $size: 0 } };
    if (tab === 'recent') {
      const last7 = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);
      match.createdAt = { $gte: last7 };
    }

    if (tag)    match.tags    = tag;
    if (listId) match.listIds = toObjId(listId);

    if (search) {
      match.$or = [
        { firstName: { $regex: search, $options: 'i' } },
        { lastName:  { $regex: search, $options: 'i' } },
        { email:     { $regex: search, $options: 'i' } },
      ];
    }

    const skip  = (Number(page) - 1) * Number(limit);
    const total = await Contact.countDocuments(match);

    const contacts = await Contact.find(match)
      .populate('listIds', 'name color')
      .sort({ createdAt: -1 })
      .skip(skip)
      .limit(Number(limit))
      .lean();

    res.json({
      success: true,
      data: contacts,
      pagination: {
        page: Number(page),
        limit: Number(limit),
        total,
        pages: Math.ceil(total / Number(limit)),
      },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── GET SINGLE CONTACT ────────────────────────────────────
exports.getContact = async (req, res) => {
  try {
    const orgId   = toObjId(req.user.orgId);
    const contact = await Contact.findOne({ _id: req.params.id, orgId })
      .populate('listIds', 'name color');
    if (!contact) return res.status(404).json({ success: false, message: 'Contact not found' });
    res.json({ success: true, data: contact });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── CREATE CONTACT ────────────────────────────────────────
exports.createContact = async (req, res) => {
  try {
    const orgId = toObjId(req.user.orgId);
    const { firstName, lastName, email, phone, tags, listIds, isVip, company, location, notes } = req.body;

    const contact = await Contact.create({
      orgId, firstName, lastName, email, phone,
      tags: tags || [], listIds: listIds || [],
      isVip: isVip || false, company, location, notes,
      source: 'manual',
    });

    if (listIds && listIds.length) {
      await ContactList.updateMany({ _id: { $in: listIds } }, { $inc: { contactCount: 1 } });
    }

    res.status(201).json({ success: true, data: contact });
  } catch (err) {
    if (err.code === 11000) return res.status(409).json({ success: false, message: 'Email already exists' });
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── UPDATE CONTACT ────────────────────────────────────────
exports.updateContact = async (req, res) => {
  try {
    const orgId   = toObjId(req.user.orgId);
    const allowed = ['firstName', 'lastName', 'phone', 'tags', 'listIds', 'isVip',
                     'isSubscribed', 'emailStatus', 'company', 'location', 'notes'];
    const updates = {};
    allowed.forEach(k => { if (req.body[k] !== undefined) updates[k] = req.body[k]; });

    const contact = await Contact.findOneAndUpdate(
      { _id: req.params.id, orgId },
      { $set: updates },
      { new: true, runValidators: true }
    ).populate('listIds', 'name color');

    if (!contact) return res.status(404).json({ success: false, message: 'Contact not found' });
    res.json({ success: true, data: contact });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── DELETE CONTACT ────────────────────────────────────────
exports.deleteContact = async (req, res) => {
  try {
    const orgId   = toObjId(req.user.orgId);
    const contact = await Contact.findOneAndDelete({ _id: req.params.id, orgId });
    if (!contact) return res.status(404).json({ success: false, message: 'Contact not found' });

    if (contact.listIds.length) {
      await ContactList.updateMany({ _id: { $in: contact.listIds } }, { $inc: { contactCount: -1 } });
    }
    res.json({ success: true, message: 'Contact deleted' });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

exports.bulkDeleteContacts = async (req, res) => {

exports.importContacts = async (req, res) => {
  try {
    const orgId = toObjId(req.user.orgId);
    const { contacts } = req.body;
    if (!contacts || !contacts.length) {
      return res.status(400).json({ success: false, message: 'No contacts provided' });
    }
    const docs = contacts.map(c => ({ ...c, orgId, source: 'csv', tags: c.tags || [], listIds: [] }));
    const result = await Contact.insertMany(docs, { ordered: false });
    res.json({ success: true, imported: result.length });
  } catch (err) {
    const inserted = err.result?.nInserted || 0;
    const skipped  = err.writeErrors?.length || 0;
    res.json({ success: true, imported: inserted, skipped, message: `${skipped} duplicates skipped` });
  }
};

// ── EXPORT CSV ────────────────────────────────────────────
exports.exportContacts = async (req, res) => {
  try {
    const orgId    = toObjId(req.user.orgId);
    const contacts = await Contact.find({ orgId }).lean();
    const header   = 'First Name,Last Name,Email,Phone,Tags,VIP,Email Status,Created\n';
    const rows     = contacts.map(c =>
      [c.firstName, c.lastName, c.email, c.phone,
       (c.tags || []).join('|'), c.isVip ? 'Yes' : 'No',
       c.emailStatus, new Date(c.createdAt).toLocaleDateString()].join(',')
    ).join('\n');
    res.setHeader('Content-Type', 'text/csv');
    res.setHeader('Content-Disposition', 'attachment; filename="contacts.csv"');
    res.send(header + rows);
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

import React, { useState, useRef } from 'react';

const ImportCSVModal = ({ isOpen, onClose, onImport }) => {
  const fileRef               = useRef();
  const [preview, setPreview] = useState([]);
  const [parsed, setParsed]   = useState([]);
  const [importing, setImporting] = useState(false);
  const [result, setResult]   = useState(null);
  const [error, setError]     = useState('');

  const parseCSV = (text) => {
    const lines   = text.trim().split('\n');
    const headers = lines[0].split(',').map(h => h.trim().toLowerCase().replace(/\s/g,''));
    return lines.slice(1).map(line => {
      const vals = line.split(',');
      const obj  = {};
      headers.forEach((h, i) => { obj[h] = (vals[i] || '').trim(); });
      return { firstName: obj.firstname || '', lastName: obj.lastname || '', email: obj.email || '', phone: obj.phone || '', tags: obj.tags ? obj.tags.split('|').map(t => t.trim()).filter(Boolean) : [] };
    }).filter(r => r.email);
  };

  const handleFile = (e) => {
    const file = e.target.files[0];
    if (!file) return;
    setResult(null); setError('');
    const reader = new FileReader();
    reader.onload = (evt) => {
      try { const rows = parseCSV(evt.target.result); setParsed(rows); setPreview(rows.slice(0,5)); }
      catch { setError('Could not parse CSV. Make sure it has headers: firstName, lastName, email, phone, tags'); }
    };
    reader.readAsText(file);
  };

  const handleImport = async () => {
    if (!parsed.length) return;
    setImporting(true); setError('');
    try { const res = await onImport(parsed); setResult(res); }
    catch (err) { setError(err.response?.data?.message || 'Import failed'); }
    finally { setImporting(false); }
  };

  const handleClose = () => { setParsed([]); setPreview([]); setResult(null); setError(''); onClose(); };

  if (!isOpen) return null;

  return (
    <div className="modal-overlay" onClick={handleClose}>
      <div className="modal-box" onClick={e => e.stopPropagation()} style={{ maxWidth: 560 }}>
        <div className="modal-header">
          <h3>Import CSV</h3>
          <button className="modal-close" onClick={handleClose}>✕</button>
        </div>
        <div className="modal-body">
          <div style={{ border: '1.5px dashed var(--border)', borderRadius: 8, padding: 24, textAlign: 'center', cursor: 'pointer', marginBottom: 16 }} onClick={() => fileRef.current?.click()}>
            <input ref={fileRef} type="file" accept=".csv" style={{ display: 'none' }} onChange={handleFile} />
            <div style={{ fontSize: 13, color: 'var(--text-muted)' }}>Click to select a <strong>.csv</strong> file</div>
            <div style={{ fontSize: 12, color: 'var(--text-muted)', marginTop: 4 }}>Columns: firstName, lastName, email, phone, tags (pipe-separated)</div>
          </div>
          {error && <div className="alert alert-error" style={{ marginBottom: 12 }}>{error}</div>}
          {preview.length > 0 && !result && (
            <div>
              <div style={{ fontSize: 13, fontWeight: 600, marginBottom: 8 }}>Preview ({parsed.length} contacts found)</div>
              <table style={{ width: '100%', fontSize: 12, borderCollapse: 'collapse' }}>
                <thead><tr style={{ background: 'var(--bg-secondary)' }}>
                  {['Name','Email','Phone','Tags'].map(h => <th key={h} style={{ padding: '6px 8px', textAlign: 'left', fontWeight: 600 }}>{h}</th>)}
                </tr></thead>
                <tbody>
                  {preview.map((r,i) => (
                    <tr key={i} style={{ borderBottom: '1px solid var(--border)' }}>
                      <td style={{ padding: '6px 8px' }}>{r.firstName} {r.lastName}</td>
                      <td style={{ padding: '6px 8px' }}>{r.email}</td>
                      <td style={{ padding: '6px 8px' }}>{r.phone}</td>
                      <td style={{ padding: '6px 8px' }}>{r.tags.join(', ')}</td>
                    </tr>
                  ))}
                </tbody>
              </table>
              {parsed.length > 5 && <div style={{ fontSize: 12, color: 'var(--text-muted)', marginTop: 6 }}>…and {parsed.length - 5} more</div>}
            </div>
          )}
          {result && <div className="alert alert-success">✅ Imported <strong>{result.imported}</strong> contacts.{result.skipped > 0 && ` ${result.skipped} duplicates skipped.`}</div>}
        </div>
        <div className="modal-footer">
          <button className="btn btn-ghost" onClick={handleClose}>{result ? 'Close' : 'Cancel'}</button>
          {!result && <button className="btn btn-primary" onClick={handleImport} disabled={!parsed.length || importing}>{importing ? 'Importing…' : `Import ${parsed.length || ''} Contacts`}</button>}
        </div>
      </div>
    </div>
  );
};

Create ProvaCampaign_BE/services/statsService.js:

js
/**
 * statsService.js
 * Called whenever an email event occurs (send, open, click, bounce).
 * Recomputes stats for ONE campaign and writes them back to Campaign.stats.
 * This keeps Reports fast — no aggregation on page load.
 */
const Campaign = require('../models/Campaign');
const EmailLog = require('../models/EmailLog');
const Event    = require('../models/Event');

/**
 * recomputeCampaignStats(campaignId)
 * Runs lightweight queries on EmailLog + Event for one campaign only.
 * Call this after any tracking event fires.
 */
exports.recomputeCampaignStats = async (campaignId) => {
  const [logs, opens, clicks, bounces, unsubs] = await Promise.all([
    EmailLog.countDocuments({ campaignId }),
    Event.countDocuments({ campaignId, type: 'open',        isFirstOccurrence: true }),
    Event.countDocuments({ campaignId, type: 'click',       isFirstOccurrence: true }),
    Event.countDocuments({ campaignId, type: 'bounce' }),
    Event.countDocuments({ campaignId, type: 'unsubscribe' }),
  ]);

  const delivered = logs - bounces;
  const base      = delivered || 1; // avoid divide-by-zero

  const stats = {
    totalSent:         logs,
    totalDelivered:    delivered,
    totalBounced:      bounces,
    totalOpens:        opens,
    totalClicks:       clicks,
    totalUnsubscribes: unsubs,
    openRate:          Math.round((opens   / base) * 1000) / 10,
    clickRate:         Math.round((clicks  / base) * 1000) / 10,
    bounceRate:        Math.round((bounces / (logs || 1)) * 1000) / 10,
    unsubscribeRate:   Math.round((unsubs  / base) * 1000) / 10,
    lastEventAt:       new Date(),
  };

  await Campaign.findByIdAndUpdate(campaignId, { $set: { stats } });
  return stats;
};

const mongoose = require('mongoose');

const toObjId = (id) => new mongoose.Types.ObjectId(id);

// ── SUMMARY ──────────────────────────────────────────────────────────────────
// Reads pre-computed stats from Campaign.stats — NO aggregation
exports.getSummary = async (req, res) => {
  try {
    const orgId     = toObjId(req.user.orgId);
    const campaigns = await Campaign.find({ orgId, 'stats.totalSent': { $gt: 0 } })
      .select('stats').lean();

    const totals = campaigns.reduce((acc, c) => {
      acc.sent        += c.stats.totalSent        || 0;
      acc.delivered   += c.stats.totalDelivered   || 0;
      acc.opens       += c.stats.totalOpens       || 0;
      acc.clicks      += c.stats.totalClicks      || 0;
      acc.bounces     += c.stats.totalBounced     || 0;
      acc.unsubs      += c.stats.totalUnsubscribes|| 0;
      return acc;
    }, { sent: 0, delivered: 0, opens: 0, clicks: 0, bounces: 0, unsubs: 0 });

    const base = totals.delivered || 1;

    res.json({
      success: true,
      data: {
        totalSent:        totals.sent,
        totalDelivered:   totals.delivered,
        openRate:         Math.round((totals.opens   / base) * 1000) / 10,
        clickRate:        Math.round((totals.clicks  / base) * 1000) / 10,
        bounceRate:       Math.round((totals.bounces / (totals.sent || 1)) * 1000) / 10,
        unsubscribeRate:  Math.round((totals.unsubs  / base) * 1000) / 10,
      },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

exports.getCampaigns = async (req, res) => {
  try {
    const orgId = toObjId(req.user.orgId);
    const { page = 1, limit = 10, status } = req.query;

    const match = { orgId };
    if (status) match.status = status;

    const skip     = (Number(page) - 1) * Number(limit);
    const total    = await Campaign.countDocuments(match);
    const campaigns = await Campaign.find(match)
      .select('name status stats createdAt scheduledAt sentAt')
      .sort({ createdAt: -1 })
      .skip(skip)
      .limit(Number(limit))
      .lean();

    res.json({
      success: true,
      data: campaigns,
      pagination: { page: Number(page), total, pages: Math.ceil(total / Number(limit)) },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

exports.getOpensOverTime = async (req, res) => {
  try {
    const orgId  = toObjId(req.user.orgId);
    const days   = Number(req.query.days || 30);
    const from   = new Date(Date.now() - days * 86400000);

    const data = await Event.aggregate([
      { $match: { orgId, type: 'open', isFirstOccurrence: true, occurredAt: { $gte: from } } },
      {
        $group: {
          _id: { $dateToString: { format: '%Y-%m-%d', date: '$occurredAt' } },
          opens: { $sum: 1 },
        },
      },
      { $sort: { _id: 1 } },
      { $project: { date: '$_id', opens: 1, _id: 0 } },
    ]);

    res.json({ success: true, data });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

exports.getCampaignDetail = async (req, res) => {
  try {
    const orgId    = toObjId(req.user.orgId);
    const campaign = await Campaign.findOne({ _id: req.params.id, orgId }).lean();
    if (!campaign) return res.status(404).json({ success: false, message: 'Campaign not found' });

    // Recipient-level history from EmailLog + Events
    const EmailLog = require('../models/EmailLog');
    const logs = await EmailLog.find({ campaignId: campaign._id })
      .populate('contactId', 'email firstName lastName')
      .lean();

    // Get all events for this campaign
    const events = await Event.find({ campaignId: campaign._id }).lean();
    const eventMap = {};
    events.forEach(e => {
      const key = e.contactId?.toString();
      if (!eventMap[key]) eventMap[key] = { opened: false, clicked: false, bounced: false, unsubscribed: false };
      if (e.type === 'open')        eventMap[key].opened      = true;
      if (e.type === 'click')       eventMap[key].clicked     = true;
      if (e.type === 'bounce')      eventMap[key].bounced     = true;
      if (e.type === 'unsubscribe') eventMap[key].unsubscribed= true;
    });

    const history = logs.map(log => {
      const contactId = log.contactId?._id?.toString() || '';
      const ev = eventMap[contactId] || {};
      return {
        email:        log.contactId?.email || log.email || '—',
        name:         log.contactId ? `${log.contactId.firstName} ${log.contactId.lastName}`.trim() : '—',
        status:       log.status || 'sent',
        opened:       ev.opened      || false,
        clicked:      ev.clicked     || false,
        bounced:      ev.bounced     || false,
        unsubscribed: ev.unsubscribed|| false,
        sentAt:       log.createdAt,
      };
    });

    res.json({
      success: true,
      data: {
        campaign,
        stats:   campaign.stats || {},
        history,
      },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

import React, { useState, useEffect, useCallback } from 'react';
import ReportSummaryCards   from '../components/reports/ReportSummaryCards';
import ReportCampaignTable  from '../components/reports/ReportCampaignTable';
import StatusPieChart       from '../components/reports/StatusPieChart';
import OpensChart           from '../components/reports/OpensChart';
import CampaignDrillDown    from '../components/reports/CampaignDrillDown';
import { fetchSummary, fetchCampaigns, fetchStatusBreakdown, fetchOpensOverTime } from '../services/reportService';

const ReportsPage = () => {
  const [summary,    setSummary]    = useState(null);
  const [campaigns,  setCampaigns]  = useState([]);
  const [pagination, setPagination] = useState(null);
  const [statusData, setStatusData] = useState([]);
  const [opensData,  setOpensData]  = useState([]);
  const [loading,    setLoading]    = useState(true);
  const [page,       setPage]       = useState(1);
  const [selectedId, setSelectedId] = useState(null); // drill-down

  const load = useCallback(async () => {
    setLoading(true);
    try {
      const [sum, camps, status, opens] = await Promise.all([
        fetchSummary(),

        fetchStatusBreakdown(),
        fetchOpensOverTime(30),

      setPagination(camps.pagination);
      setStatusData(status || []);
      setOpensData(opens || []);

    } finally {
      setLoading(false);
    }

  useEffect(() => { load(); }, [load]);

  // Drill-down view
  if (selectedId) {
    return (
      <div className="page-container">
        <CampaignDrillDown campaignId={selectedId} onBack={() => setSelectedId(null)} />
      </div>

      </div>

      {/* 2. Campaign Performance Table */}
      <ReportCampaignTable
        campaigns={campaigns}
        loading={loading}
        pagination={pagination}
        onSelect={setSelectedId}
        onPageChange={setPage}
      />

      {/* 3. Charts Row */}
      <div style={{ display: 'grid', gridTemplateColumns: '1fr 2fr', gap: 16 }}>
        <StatusPieChart data={statusData} loading={loading} />
        <OpensChart     data={opensData}  loading={loading} />
      </div>
    </div>
  );
};

const Campaign = require('../models/Campaign');

// ── SUMMARY ──────────────────────────────────────────────────────────────────
exports.getSummary = async (req, res) => {
  try {
    const campaigns = await Campaign.find({ 'stats.totalSent': { $gt: 0 } })
      .select('stats').lean();

    const totals = campaigns.reduce((acc, c) => {
      acc.sent   += c.stats.totalSent         || 0;
      acc.delivered += c.stats.totalDelivered || 0;
      acc.opens  += c.stats.totalOpens        || 0;
      acc.clicks += c.stats.totalClicks       || 0;
      acc.bounces+= c.stats.totalBounced      || 0;
      acc.unsubs += c.stats.totalUnsubscribes || 0;
      return acc;
    }, { sent: 0, delivered: 0, opens: 0, clicks: 0, bounces: 0, unsubs: 0 });

    const base = totals.delivered || 1;

    res.json({
      success: true,
      data: {
        totalSent:       totals.sent,
        totalDelivered:  totals.delivered,
        openRate:        Math.round((totals.opens   / base)             * 1000) / 10,
        clickRate:       Math.round((totals.clicks  / base)             * 1000) / 10,
        bounceRate:      Math.round((totals.bounces / (totals.sent||1)) * 1000) / 10,
        unsubscribeRate: Math.round((totals.unsubs  / base)             * 1000) / 10,
      },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── CAMPAIGN TABLE ────────────────────────────────────────────────────────────
exports.getCampaigns = async (req, res) => {
  try {
    const { page = 1, limit = 10, status } = req.query;
    const match = {};
    if (status) match.status = status;

    const skip  = (Number(page) - 1) * Number(limit);
    const total = await Campaign.countDocuments(match);
    const campaigns = await Campaign.find(match)
      .select('name status stats createdAt publishedAt')
      .sort({ createdAt: -1 })
      .skip(skip)
      .limit(Number(limit))
      .lean();

    res.json({
      success: true,
      data: campaigns,
      pagination: { page: Number(page), total, pages: Math.ceil(total / Number(limit)) },
    });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── CAMPAIGN STATUS PIE ───────────────────────────────────────────────────────
exports.getCampaignStatusBreakdown = async (req, res) => {
  try {
    const breakdown = await Campaign.aggregate([
      { $group: { _id: '$status', count: { $sum: 1 } } },
    ]);
    res.json({ success: true, data: breakdown });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

// ── OPENS OVER TIME ───────────────────────────────────────────────────────────
exports.getOpensOverTime = async (req, res) => {
  try {
    const days = Number(req.query.days || 30);
    const from = new Date(Date.now() - days * 86400000);

    const data = await Event.aggregate([
      { $match: { type: 'open', isFirstOccurrence: true, occurredAt: { $gte: from } } },
      {
        $group: {
          _id:   { $dateToString: { format: '%Y-%m-%d', date: '$occurredAt' } },
          opens: { $sum: 1 },
        },
      },
      { $sort: { _id: 1 } },
      { $project: { date: '$_id', opens: 1, _id: 0 } },
    ]);

    res.json({ success: true, data });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
};

exports.getCampaignDetail = async (req, res) => {
  try {
    const campaign = await Campaign.findById(req.params.id).lean();
    if (!campaign) return res.status(404).json({ success: false, message: 'Campaign not found' });

    // Use correct field name: campaign (not campaignId)
    const logs = await EmailLog.find({ campaign: campaign._id }).lean();

    const events = await Event.find({ campaign: campaign._id }).lean();

    // Map events by emailLog id
    const eventMap = {};
    events.forEach(e => {
      const key = e.emailLog?.toString();
      if (!eventMap[key]) eventMap[key] = { opened: false, clicked: false, bounced: false, unsubscribed: false };
      if (e.type === 'open')        eventMap[key].opened       = true;
      if (e.type === 'click')       eventMap[key].clicked      = true;
      if (e.type === 'bounce')      eventMap[key].bounced      = true;
      if (e.type === 'unsubscribe') eventMap[key].unsubscribed = true;
    });

    const history = logs.map(log => {
      const ev = eventMap[log._id?.toString()] || {};
      return {
        email:        log.recipientEmail || '—',
        name:         log.recipientName  || '—',
        status:       log.status         || 'sent',
        opened:       ev.opened          || false,
        clicked:      ev.clicked         || false,
        bounced:      ev.bounced         || false,
        unsubscribed: ev.unsubscribed    || false,
        sentAt:       log.sentAt,
      };
    });

    res.json({
      success: true,
      data: { campaign, stats: campaign.stats || {}, history },
    });
